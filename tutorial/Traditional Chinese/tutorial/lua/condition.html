判斷式
判斷式也稱為條件式
用來進行邏輯判斷
程式非常需要這些判斷
例如下面情境


-- 一個用來計算並印出倒數的函式
local reciprocal = function(number)
    print( 1/number )
end


-- 這樣的函式會有問題

reciprocal(1)   -- 印出 1
reciprocal(0.5) -- 印出 2
reciprocal(0.1) -- 印出 10
reciprocal(0)   -- 0 是無法計算倒數的，必須告知使用者這問題

-- 邏輯上應該要這麼寫才對
--[[
local reciprocal = function(number)
    如果 number 等於0 就印出 "You can't do this!" 訊息來提醒使用者
    0 以外的情況就計算 number 的倒數
end
]]

-- 寫成程式就是這樣
local reciprocal = function(number)
    if number==0 then
        print( "You can't do this!" )
    else
        print( 1/number )
    end
end



























各種邏輯判斷


if     number==0 then end  -- 如果 number 等於0
if     number~=0 then end  -- 如果 number 不等於0
if not number==0 then end  -- 如果 number 等於0 並非事實
if not number~=0 then end  -- 如果 number 不等於0 並非事實


-- 很顯然的
number==0
-- 跟
not number~=0
-- 是一樣的意思

number~=0
-- 跟
not number==0
-- 也是一樣的意思


















多條件判斷


-- 如果 x,y,z 都是0
if x==0 then
    if y==0 then
        if z==0 then
            print("(x,y,z)=(0,0,0)")
        end
    end
end

if x==0 and y==0 and z==0 then end   -- 如果 x,y,z 都是0
if x==0 or  y==0 or  z==0 then end   -- 如果 x,y,z 其中有一個等於0













其實每一樣條件都各別用一個if來判斷也可以
選你覺得看起來比較舒服的寫法就好了
